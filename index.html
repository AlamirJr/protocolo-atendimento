Júnior # ============================================================ !pip -q install openpyxl ipywidgets pandas import os import re import openpyxl import pandas as pd from datetime import datetime import ipywidgets as widgets from IPython.display import display, clear_output # ============================================================ # 1) GOOGLE DRIVE + CAMINHO DO ARQUIVO # ============================================================ from google.colab import drive drive.mount("/content/drive") EXCEL_PATH = "/content/drive/MyDrive/automacoes_inclusiva/ProtocoloAtendimento/atendimento.xlsx" SHEET_NAME = "Atendimentos" MAP_SHEET = "ID_MAP" EXPECTED_HEADERS = [ "ID","Data/Hora","Atendimento","Telefone","Vínculo","Aluno","Idade", "AE","Turno","PSP Responsável","Solicitação","Tipo","Status" ] PREFIX = "GEE" SEQ_WIDTH = 6 # GEE-2026-000001 print("✅ Arquivo:", EXCEL_PATH) print("📌 Existe?", os.path.exists(EXCEL_PATH)) # ============================================================ # 2) FUNÇÕES BÁSICAS # ============================================================ def load_wb(): if not os.path.exists(EXCEL_PATH): raise FileNotFoundError(f"Arquivo não encontrado: {EXCEL_PATH}") wb = openpyxl.load_workbook(EXCEL_PATH) if SHEET_NAME not in wb.sheetnames: raise ValueError(f"Aba '{SHEET_NAME}' não existe. Abas disponíveis: {wb.sheetnames}") ws = wb[SHEET_NAME] return wb, ws def validate_or_fix_headers(ws): current = [ws.cell(1, c).value for c in range(1, len(EXPECTED_HEADERS) + 1)] if all(v is None or str(v).strip() == "" for v in current): for c, h in enumerate(EXPECTED_HEADERS, start=1): ws.cell(1, c).value = h return if current != EXPECTED_HEADERS: raise ValueError( "Cabeçalhos da planilha não batem com o esperado.\n" f"Esperado: {EXPECTED_HEADERS}\n" f"Encontrado: {current}" ) def normalize_phone(s: str) -> str: if not s: return "" return re.sub(r"\D", "", str(s)) # ============================================================ # 3) ID INSTITUCIONAL # ============================================================ def is_institutional_id(pid: str): if not pid: return False s = str(pid).strip() return re.fullmatch(rf"{re.escape(PREFIX)}-\d{{4}}-\d{{{SEQ_WIDTH}}}", s) is not None def parse_seq_from_id(pid: str, year: int): if not pid: return None s = str(pid).strip() m = re.fullmatch(rf"{re.escape(PREFIX)}-{year}-(\d{{{SEQ_WIDTH}}})", s) if not m: return None return int(m.group(1)) def next_institutional_id(ws, year: int): max_seq = 0 for r in range(2, ws.max_row + 1): v = ws.cell(r, 1).value seq = parse_seq_from_id(v, year) if seq is not None: max_seq = max(max_seq, seq) new_seq = max_seq + 1 return f"{PREFIX}-{year}-{new_seq:0{SEQ_WIDTH}d}" def get_year_from_datetime_cell(v): if isinstance(v, datetime): return v.year s = str(v).strip() m = re.search(r"(\d{2})/(\d{2})/(\d{4})", s) if m: return int(m.group(3)) return datetime.now().year # ============================================================ # 4) CRUD # ============================================================ def append_attendance(values_by_header: dict): wb, ws = load_wb() validate_or_fix_headers(ws) year = datetime.now().year new_id = next_institutional_id(ws, year) now_str = datetime.now().strftime("%d/%m/%Y %H:%M:%S") row = [ new_id, now_str, values_by_header.get("Atendimento","").strip(), normalize_phone(values_by_header.get("Telefone","")), values_by_header.get("Vínculo",""), values_by_header.get("Aluno","").strip(), values_by_header.get("Idade",""), values_by_header.get("AE",""), values_by_header.get("Turno",""), values_by_header.get("PSP Responsável","").strip(), values_by_header.get("Solicitação","").strip(), values_by_header.get("Tipo",""), values_by_header.get("Status","") ] ws.append(row) wb.save(EXCEL_PATH) wb.close() return new_id, now_str def read_df(): wb, ws = load_wb() validate_or_fix_headers(ws) data = list(ws.values) wb.close() if len(data) <= 1: return pd.DataFrame(columns=EXPECTED_HEADERS) return pd.DataFrame(data[1:], columns=data[0]) def update_status_by_id(protocolo_id: str, new_status: str): wb, ws = load_wb() validate_or_fix_headers(ws) target = str(protocolo_id).strip() if not target: wb.close() raise ValueError("Informe um ID válido.") found_row = None for r in range(2, ws.max_row + 1): cell_val = ws.cell(r, 1).value if cell_val is None: continue if str(cell_val).strip() == target: found_row = r break if found_row is None: sample = [] for r in range(2, min(ws.max_row + 1, 20)): sample.append(ws.cell(r, 1).value) wb.close() raise ValueError(f"ID '{target}' não encontrado. Exemplos: {sample}") ws.cell(found_row, 13).value = new_status wb.save(EXCEL_PATH) wb.close() # ============================================================ # 5) MIGRAÇÃO DE IDs ANTIGOS # ============================================================ def ensure_map_sheet(wb): if MAP_SHEET not in wb.sheetnames: map_ws = wb.create_sheet(MAP_SHEET) map_ws.append(["Data/Hora Migração", "Linha", "ID_antigo", "ID_novo"]) else: map_ws = wb[MAP_SHEET] return map_ws def count_non_institutional_ids(): wb, ws = load_wb() validate_or_fix_headers(ws) count = 0 examples = [] for r in range(2, ws.max_row + 1): old_id = ws.cell(r, 1).value if old_id and not is_institutional_id(old_id): count += 1 if len(examples) < 10: examples.append(str(old_id)) wb.close() return count, examples def migrate_old_ids(): wb, ws = load_wb() validate_or_fix_headers(ws) map_ws = ensure_map_sheet(wb) changed = 0 for r in range(2, ws.max_row + 1): old_id = ws.cell(r, 1).value if not old_id: continue if is_institutional_id(old_id): continue year = get_year_from_datetime_cell(ws.cell(r, 2).value) new_id = next_institutional_id(ws, year) ws.cell(r, 1).value = new_id map_ws.append([ datetime.now().strftime("%d/%m/%Y %H:%M:%S"), r, str(old_id), new_id ]) changed += 1 wb.save(EXCEL_PATH) wb.close() return changed # ============================================================ # 6) INTERFACE # ============================================================ w_atendimento = widgets.Text( description="Atendimento*", placeholder="Ex.: Orientação / Encaminhamento / Registro", layout=widgets.Layout(width="720px") ) w_telefone = widgets.Text( description="Telefone", placeholder="(xx) xxxxx-xxxx", layout=widgets.Layout(width="320px") ) w_vinculo = widgets.Dropdown( description="Vínculo", options=["Responsável", "Aluno", "Escola", "Outro"], value="Responsável" ) w_aluno = widgets.Text( description="Aluno*", placeholder="Nome completo", layout=widgets.Layout(width="720px") ) w_idade = widgets.IntText( description="Idade", value=0, layout=widgets.Layout(width="180px") ) w_ae = widgets.Text( description="AE", placeholder="Ex.: Sala de Recursos / Professor(a) AE", layout=widgets.Layout(width="420px") ) w_turno = widgets.Dropdown( description="Turno", options=["Manhã", "Tarde", "Noite", "Integral", "N/I"], value="N/I" ) w_psp = widgets.Text( description="PSP Resp.", placeholder="Responsável pelo atendimento", layout=widgets.Layout(width="520px") ) w_solic = widgets.Textarea( description="Solicitação*", placeholder="Descreva a demanda, contexto, ações e encaminhamentos...", layout=widgets.Layout(width="920px", height="130px") ) w_tipo = widgets.Dropdown( description="Tipo", options=["Orientação", "Encaminhamento", "Registro", "Outros"], value="Registro" ) w_status = widgets.Dropdown( description="Status", options=["Aberto", "Em andamento", "Concluído"], value="Aberto" ) btn_save = widgets.Button(description="Salvar atendimento", button_style="success", icon="save") btn_clear = widgets.Button(description="Limpar", icon="eraser") btn_last = widgets.Button(description="Ver últimos 20", button_style="info", icon="list") w_busca = widgets.Text( description="Buscar", placeholder="Aluno / telefone / status / tipo / ID", layout=widgets.Layout(width="720px") ) btn_busca = widgets.Button(description="Pesquisar", button_style="warning", icon="search") w_upd_id = widgets.Text( description="ID", placeholder=f"Cole o ID (ex.: {PREFIX}-{datetime.now().year}-000001)", layout=widgets.Layout(width="420px") ) w_upd_status = widgets.Dropdown( description="Novo status", options=["Aberto", "Em andamento", "Concluído"], value="Em andamento" ) btn_upd = widgets.Button(description="Atualizar status", button_style="primary", icon="refresh-cw") btn_check_mig = widgets.Button(description="Verificar IDs antigos", button_style="warning", icon="search") btn_migrate = widgets.Button(description="Migrar IDs antigos", button_style="danger", icon="exchange") out = widgets.Output() # ============================================================ # 7) EVENTOS # ============================================================ def clear_form(): w_atendimento.value = "" w_telefone.value = "" w_vinculo.value = "Responsável" w_aluno.value = "" w_idade.value = 0 w_ae.value = "" w_turno.value = "N/I" w_psp.value = "" w_solic.value = "" w_tipo.value = "Registro" w_status.value = "Aberto" def on_save(_): with out: clear_output() if not w_atendimento.value.strip() or not w_aluno.value.strip() or not w_solic.value.strip(): print("⚠️ Preencha os campos obrigatórios: Atendimento, Aluno e Solicitação.") return payload = { "Atendimento": w_atendimento.value, "Telefone": w_telefone.value, "Vínculo": w_vinculo.value, "Aluno": w_aluno.value, "Idade": int(w_idade.value) if w_idade.value else "", "AE": w_ae.value, "Turno": w_turno.value, "PSP Responsável": w_psp.value, "Solicitação": w_solic.value, "Tipo": w_tipo.value, "Status": w_status.value } try: rid, now = append_attendance(payload) print(f"✅ Salvo com sucesso! ID={rid} | Data/Hora={now}") print("📌 Arquivo:", EXCEL_PATH) clear_form() except Exception as e: print("❌ Erro ao salvar:", e) def on_clear(_): with out: clear_output() clear_form() print("🧹 Formulário limpo.") print("📌 Arquivo:", EXCEL_PATH) def on_last(_): with out: clear_output() try: df = read_df() if df.empty: print("ℹ️ Não há registros ainda.") else: display(df.tail(20)) print("📌 Arquivo:", EXCEL_PATH) except Exception as e: print("❌ Erro ao ler planilha:", e) def on_search(_): with out: clear_output() q = w_busca.value.strip().lower() if not q: print("Digite algo para buscar.") return try: df = read_df().fillna("") if df.empty: print("ℹ️ Não há registros ainda.") return mask = ( df["ID"].astype(str).str.lower().str.contains(q) | df["Aluno"].astype(str).str.lower().str.contains(q) | df["Telefone"].astype(str).str.lower().str.contains(q) | df["Status"].astype(str).str.lower().str.contains(q) | df["Tipo"].astype(str).str.lower().str.contains(q) | df["Atendimento"].astype(str).str.lower().str.contains(q) | df["Solicitação"].astype(str).str.lower().str.contains(q) ) res = df[mask].sort_values(by="Data/Hora", ascending=False).head(50) if res.empty: print("Nada encontrado.") else: display(res) print("📌 Arquivo:", EXCEL_PATH) except Exception as e: print("❌ Erro na busca:", e) def on_update(_): with out: clear_output() pid = w_upd_id.value.strip() if not pid: print("⚠️ Informe um ID válido.") return try: update_status_by_id(pid, w_upd_status.value) print(f"✅ Status do ID {pid} atualizado para: {w_upd_status.value}") print("📌 Arquivo:", EXCEL_PATH) except Exception as e: print("❌ Erro ao atualizar:", e) def on_check_migration(_): with out: clear_output() try: count, examples = count_non_institutional_ids() print(f"🔎 Registros com ID antigo/fantasia: {count}") if examples: print("Exemplos:") for ex in examples: print("-", ex) else: print("Nenhum ID antigo encontrado.") print("📌 Arquivo:", EXCEL_PATH) except Exception as e: print("❌ Erro ao verificar migração:", e) def on_migrate(_): with out: clear_output() try: count_before, examples = count_non_institutional_ids() if count_before == 0: print("✅ Não há IDs antigos para migrar.") print("📌 Arquivo:", EXCEL_PATH) return print(f"🔁 Iniciando migração de {count_before} registro(s)...") changed = migrate_old_ids() print(f"✅ Migração concluída. Registros alterados: {changed}") print(f"📋 Aba de auditoria criada/atualizada: {MAP_SHEET}") print("📌 Arquivo:", EXCEL_PATH) except Exception as e: print("❌ Erro na migração:", e) btn_save.on_click(on_save) btn_clear.on_click(on_clear) btn_last.on_click(on_last) btn_busca.on_click(on_search) btn_upd.on_click(on_update) btn_check_mig.on_click(on_check_migration) btn_migrate.on_click(on_migrate) # ============================================================ # 8) LAYOUT # ============================================================ header = widgets.HTML( "

"), widgets.HBox([w_busca, btn_busca]), widgets.HTML("
"), widgets.HTML("Atualizar status por ID"), widgets.HBox([w_upd_id, w_upd_status, btn_upd]), widgets.HTML("
"), widgets.HTML("Migração de IDs antigos"), widgets.HBox([btn_check_mig, btn_migrate]), out ]) display(ui) with out: print("Pronto.") print("1) Novos registros já saem com ID institucional.") print("2) Use 'Verificar IDs antigos' para saber quantos ainda estão fora do padrão.") print("3) Use 'Migrar IDs antigos' para converter os IDs fantasia.") print("4) A aba 'ID_MAP' guarda o histórico da migração.") print("📌 Arquivo:", EXCEL_PATH)
